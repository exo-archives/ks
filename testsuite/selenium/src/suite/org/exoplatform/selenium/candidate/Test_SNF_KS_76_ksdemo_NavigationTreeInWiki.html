echo - Sign in
open /ksdemo/public/classic/
windowMaximize
waitForElementPresent link=Sign in
click link=Sign in
waitForElementPresent username
type username root
waitForElementPresent password
type password gtn
clickAndWait //div[@id='UIPortalLoginFormAction']/div/div/div[@class='ButtonMiddle']/a
open /ksdemo/private/classic/wiki
echo - Verify the initial state of the tree:
echo -- A node labeled Wiki Home is highlighted
waitForElementPresent //div[@id='portal%2Fclassic%2FWikiHome']/div[@id='iconTreeExplorer' and @class='Wikihome TreeNodeType Node Hover ']/div/a[@title='Wiki Home']
verifyText //div[@id='portal%2Fclassic%2FWikiHome']/div[@id='iconTreeExplorer' and @class='Wikihome TreeNodeType Node Hover ']/div/a[@title='Wiki Home'] Wiki Home
echo - Create a page node
waitForElementPresent link=Blank page
click link=Blank page
waitForElementPresent TitleInput
type TitleInput Page1_SNF_KS_76
waitForElementPresent Markup
type Markup Page1 Content
waitForElementPresent link=Save
click link=Save
waitForElementPresent link=Page1_SNF_KS_76
waitForTextPresent Page1 Content
verifyTextPresent Page1 Content
echo - Verify that the newly created node is childless and is highlighted
waitForElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='EmptyIcon']/div[contains(@class, 'Page TreeNodeType Node Hover')]/div/a[@title='Page1_SNF_KS_76']
verifyText //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='EmptyIcon']/div[@class='Page TreeNodeType Node Hover ']/div/a[@title='Page1_SNF_KS_76'] Page1_SNF_KS_76
echo - Create a child page node
waitForElementPresent link=Blank page
click link=Blank page
waitForElementPresent TitleInput
type TitleInput Page2_SNF_KS_76
waitForElementPresent Markup
type Markup Page2 Content
waitForElementPresent link=Save
click link=Save
waitForElementPresent link=Page2_SNF_KS_76
waitForTextPresent Page2 Content
verifyTextPresent Page2 Content
echo - Verify that your first node now have a child node and is not highlighted
waitForElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='CollapseIcon']/div[contains(@class,'Page TreeNodeType Node') and not(contains(@class,'Hover'))]/div/a[@title='Page1_SNF_KS_76']
verifyElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='CollapseIcon']/div[contains(@class,'Page TreeNodeType Node') and not(contains(@class,'Hover'))]/div/a[@title='Page1_SNF_KS_76']
verifyElementNotPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='CollapseIcon']/div[contains(@class,'Hover')]
echo - Click the icon of the parent node and verify ...
click //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76']
echo --------------- that the Page Area still shows the child page not the parent
waitForElementPresent TitleInfo
verifyText TitleInfo Page2_SNF_KS_76
waitForElementPresent //div[@id='UIWikiPageContentArea']/p
verifyText //div[@id='UIWikiPageContentArea']/p Page2 Content
echo --------------- that the node group is collapsed
waitForElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='ExpandIcon']/div[contains(@class,'Page TreeNodeType Node') and not(contains(@class,'Hover'))]/div/a[@title='Page1_SNF_KS_76']
verifyElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='ExpandIcon']/div[contains(@class,'Page TreeNodeType Node') and not(contains(@class,'Hover'))]/div/a[@title='Page1_SNF_KS_76']
verifyElementNotPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='CollapseIcon']/div[contains(@class,'Page TreeNodeType Node') and not(contains(@class,'Hover'))]/div/a[@title='Page1_SNF_KS_76']
echo - Click the link of the parent node and verify ...
click //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76']//a[@title='Page1_SNF_KS_76']
echo --------------- that the Page Area now shows the parent page
waitForText TitleInfo Page1_SNF_KS_76
verifyText TitleInfo Page1_SNF_KS_76
waitForElementPresent //div[@id='UIWikiPageContentArea']/p
verifyText //div[@id='UIWikiPageContentArea']/p Page1 Content
echo --------------- that this click also expands the group
waitForElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='CollapseIcon']
verifyElementPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='CollapseIcon']
verifyElementNotPresent //div[@id='portal%2Fclassic%2FPage1_SNF_KS_76' and @class='ExpandIcon']
echo - Delete test data
open /ksdemo/private/classic/wiki/Page2_SNF_KS_76
waitForElementPresent link=Delete page
clickAt link=Delete page
waitForElementPresent //div[@id='UIWikiPageArea']/div/div[2]/table/tbody/tr/td[1]/div/div/div/div/a
clickAndWait //div[@id='UIWikiPageArea']/div/div[2]/table/tbody/tr/td[1]/div/div/div/div/a
waitForTextNotPresent Page2_SNF_KS_76
verifyTextNotPresent Page2_SNF_KS_76
open /ksdemo/private/classic/wiki/Page1_SNF_KS_76
waitForElementPresent link=Delete page
clickAt link=Delete page
waitForElementPresent //div[@id='UIWikiPageArea']/div/div[2]/table/tbody/tr/td[1]/div/div/div/div/a
clickAndWait //div[@id='UIWikiPageArea']/div/div[2]/table/tbody/tr/td[1]/div/div/div/div/a
waitForTextNotPresent Page1_SNF_KS_76
verifyTextNotPresent Page1_SNF_KS_76

waitForElementPresent link=Sign out
clickAndWait link=Sign out
