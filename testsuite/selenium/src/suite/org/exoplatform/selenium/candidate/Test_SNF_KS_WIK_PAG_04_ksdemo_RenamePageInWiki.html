Test_SNF_KS_61_ksdemo_QuickEditTitleWiki
echo - Sign in
open /ksdemo/public/classic/
windowMaximize
waitForElementPresent link=Sign in
click link=Sign in
waitForElementPresent username
type username root
waitForElementPresent password
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
open /ksdemo/private/classic/wiki
echo - Create page
waitForElementPresent link=Blank Page
click link=Blank Page
waitForElementPresent TitleInput
type TitleInput Page1_SNF_KS_61
waitForElementPresent Markup
type Markup Page1 Content
clickAndWait link=Save
waitForTextPresent Page1_SNF_KS_61
verifyTextPresent Page1_SNF_KS_61
waitForTextPresent Page1 Content
verifyTextPresent Page1 Content
echo - Edit Title
waitForElementPresent TitleInfo
doubleClick TitleInfo
waitForElementPresent EdiableInput
type EdiableInput Page1_SNF_KS_61_Edited
waitForElementPresent //form[@id='Editable']//a[@class='SubmitLink']
click //form[@id='Editable']//a[@class='SubmitLink']
waitForTextPresent Page1_SNF_KS_61_Edited
verifyTextPresent Page1_SNF_KS_61_Edited
echo - Delete test data
open /ksdemo/private/classic/wiki/WikiHome
waitForElementPresent link=Page1_SNF_KS_61_Edited
click link=Page1_SNF_KS_61_Edited
waitForElementPresent link=Delete Page
clickAt link=Delete Page
waitForElementPresent //div[@id='UIWikiPageArea']//a[text()='OK']
clickAndWait //div[@id='UIWikiPageArea']//a[text()='OK']
waitForTextNotPresent link=Page1_SNF_KS_61_Edited
verifyTextNotPresent link=Page1_SNF_KS_61_Edited
echo - Sign out and finish
clickAndWait link=Sign out
waitForTextPresent Sign in
echo - End of test
