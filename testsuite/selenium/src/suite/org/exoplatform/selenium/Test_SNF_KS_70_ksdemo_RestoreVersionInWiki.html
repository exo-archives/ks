echo - Sign in
open /ksdemo/public/classic/
windowMaximize
waitForElementPresent link=Sign in
click link=Sign in
waitForElementPresent username
type username root
waitForElementPresent password
type password gtn
clickAndWait //div[@id='UIPortalLoginFormAction']/div/div/div[@class='ButtonMiddle']/a
open /ksdemo/private/classic/wiki
echo - Create the 1st revision
waitForElementPresent link=add
click link=add
waitForElementPresent TitleInput
type TitleInput Page1_SNF_KS_70
waitForElementPresent Markup
type Markup The 1st revision
clickAndWait link=save
waitForTextPresent Page1_SNF_KS_70
verifyTextPresent Page1_SNF_KS_70
waitForTextPresent The 1st revision
verifyTextPresent The 1st revision
echo - Create the 2nd revision
waitForElementPresent link=edit
click link=edit
waitForElementPresent TitleInput
type TitleInput Page1_SNF_KS_70
waitForElementPresent Markup
type Markup The 2nd revision
clickAndWait link=save
waitForText TitleInfo Page1_SNF_KS_70
verifyText TitleInfo Page1_SNF_KS_70
waitForText //div[@id='UIWikiPageContentArea']/p The 2nd revision
verifyText //div[@id='UIWikiPageContentArea']/p The 2nd revision
echo - Restore to the 1st revision
waitForTextPresent 2 revisions
verifyTextPresent 2 revisions
waitForElementPresent link=(view change)
click link=(view change)
waitForElementPresent link=Restore
click link=Restore
waitForTextPresent The 1st revision
verifyTextPresent The 1st revision
waitForTextPresent 3 revisions
verifyTextPresent 3 revisions
echo - Delete test data
waitForElementPresent link=delete
clickAt link=delete
waitForElementPresent //div[@id='UIWikiPageArea']/div/div[2]/table/tbody/tr/td[1]/div/div/div/div/a
clickAndWait //div[@id='UIWikiPageArea']/div/div[2]/table/tbody/tr/td[1]/div/div/div/div/a
waitForTextNotPresent Page1_SNF_KS_70
verifyTextNotPresent Page1_SNF_KS_70
echo - Sign out and finish
waitForElementPresent link=Sign out
clickAndWait link=Sign out
waitForTextPresent Sign in
echo - End of test
